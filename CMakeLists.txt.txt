# Specify the minimum version of CMake required
cmake_minimum_required(VERSION 3.14...3.16)

# Set the project name
project(SMAT-ADT
	VERSION 0.4.1
	DESCRIPTION "Tool"
	LANGUAGES CXX)

if(CMAKE_PROJECT_NAME STREQUAL PROJECT_NAME)
	set_property(GLOBAL PROPERTY USE_FOLDERS ON)
	include(CTest)
endif()

include(FetchContent)

add_subdirectory(src)

# Add the executable target
add_executable(tool_gui ${GUI_TYPE}
		apps/main.cpp
		${APP_ICON_RESOURCE_WINDOWS}
		)

configure_file(apps/version.h.in version.h)

# In order to run as debug a console is needed
IF(WIN32)
	set_target_properties(tool_gui PROPERTIES COMPILE_DEFINITIONS_DEBUG "_CONSOLE")
ENDIF(WIN32)

target_compile_features(smat_gui PRIVATE cxx_std_11)
target_link_libraries(smat_gui PRIVATE qt_gui)
